
public class DollarPattern 
{
	public static void main(String []arg)
	{
		int i,j;
		System.out.println("$ $ $ $ ");
		for(i=1;i<=2;i++)
		{
			for(j=i;j<=i;j++)
			{
				System.out.println("$     $");
			}}
			System.out.println("$ $ $ $");
		
		
	}

}
